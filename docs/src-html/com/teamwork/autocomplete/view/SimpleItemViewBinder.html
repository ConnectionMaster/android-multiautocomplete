017package com.teamwork.autocomplete.view;
018
019import android.view.View;
020import android.widget.ImageView;
021import android.widget.TextView;
022
023import androidx.annotation.NonNull;
024import androidx.annotation.Nullable;
025
026import com.teamwork.autocomplete.R;
027import com.teamwork.autocomplete.model.SimpleItem;
028import com.teamwork.autocomplete.util.SpannableUtils;
029
030/**
031 * Item view binder for a {@link SimpleItem} model.
032 * <p>
033 * It provides very basic behaviour for components that don't need any customization:
034 * <ul>
035 * <li>{@link #getItemId(SimpleItem)} returns the item's hash code</li>
036 * <li>It binds the data to the R.layout.simple_item layout</li>
037 * <li>It highlights the portion of text that matches the current text constraint</li>
038 * </ul>
039 * To prevent the library to have dependencies on any image loading/caching framework, images are not loaded into the
040 * image view. To implement your own loading, override {@link #onImageViewBind(ImageView, SimpleItem)}.
041 *
042 * @author Marco Salis
043 */
044@SuppressWarnings("WeakerAccess")
045public class SimpleItemViewBinder implements AutoCompleteViewBinder<SimpleItem> {
046
047    @Override
048    public long getItemId(@NonNull SimpleItem item) {
049        return item.hashCode();
050    }
051
052    @Override
053    public int getItemLayoutId() {
054        return R.layout.simple_item;
055    }
056
057    @Override
058    public @NonNull AutoCompleteViewHolder getViewHolder(@NonNull View view) {
059        return new SimpleItemViewHolder(view);
060    }
061
062    @Override
063    public void bindData(@NonNull AutoCompleteViewHolder viewHolder, @NonNull SimpleItem item, @Nullable CharSequence constraint) {
064        SimpleItemViewHolder itemViewHolder = (SimpleItemViewHolder) viewHolder;
065        CharSequence text;
066        if (constraint != null) { // highlight constraint text in bold if any
067            text = SpannableUtils.setBoldSubText(item.getText(), constraint);
068        } else {
069            text = item.getText();
070        }
071        itemViewHolder.textView.setText(text);
072
073        onImageViewBind(itemViewHolder.imageView, item);
074    }
075
076    /**
077     * Override this to provide image loading to the {@link SimpleItemViewBinder} using your own image caching library.
078     * <p>
079     * The default implementation just sets the ImageView's visibility to {@link View#GONE}.
080     */
081    protected void onImageViewBind(@NonNull ImageView imageView, @SuppressWarnings("unused") @NonNull SimpleItem item) {
082        imageView.setVisibility(View.GONE);
083    }
084
085
086    public static class SimpleItemViewHolder extends AutoCompleteViewHolder {
087
088        public final ImageView imageView;
089        public final TextView textView;
090
091        public SimpleItemViewHolder(@NonNull View view) {
092            super(view);
093            imageView = view.findViewById(R.id.imageView);
094            textView = view.findViewById(R.id.textView);
095        }
096    }
097
098}
